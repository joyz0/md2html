- 与或运算来做类型判断

```js
const A = 0b0000000000000001;
const B = 0b0000000000000010;
const C = 0b0000000000000100;
const D = 0b0000000000001000;
const E = 0b0000000000010000;

const AB = A | B;
const CD = C | D;

let current = random();
if (current & AB) {
  console.log('属于AB');
} else if (current & CD) {
  console.log('属于CD');
} else {
  console.log('属于E');
}
if (current > C) {
  console.log('在C阶段之后');
}
function random() {
  const pool = [A, B, C, D, E];
  return pool[Math.floor(5 * Math.random())];
}
```
